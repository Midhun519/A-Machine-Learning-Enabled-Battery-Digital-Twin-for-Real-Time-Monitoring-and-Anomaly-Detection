<script>
    var socket = io();

    socket.on("update", function(data) {
        document.getElementById("voltage").innerHTML = data.voltage.toFixed(3);
        document.getElementById("temp").innerHTML = data.temperature.toFixed(2);
        document.getElementById("current").innerHTML = data.current.toFixed(3);

        let status = document.getElementById("status");
        status.innerHTML = data.status;

        if (data.status === "NORMAL") {
            status.className = "status normal";
        } else {
            status.className = "status anomaly";
        }
    });
</script>
